7. "node-sass": "^4.11.0",
8. "postcss-loader": "^3.0.0",
9. "sass-loader": "^7.1.0",
10. "style-loader": "^0.23.1",
11. "url-loader": "^1.1.2",
12. "webpack": "^4.27.1"
